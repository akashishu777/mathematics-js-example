## Installation

### NPM

```sh
npm install mathematics-js-example --save
```


## Usage
```js
const math = require('mathematics-js-example');

let answer1 = math.sqr(5);
console.log( 'sqr Answer: ' + answer1 );

let answer2 = math.add(5,5);
console.log( 'add Answer: ' + answer2 );
```
